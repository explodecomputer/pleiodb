Overview

We evaluate how well the elastic-net LD imputation recovers masked z-scores in two contrasting genomic settings:

  1. Null region — chromosome 1 block 168414490–169838848, BMI GWAS (ukb-b-19953). No genome-wide significant variants; z-scores are noise.
  2. Signal region — chromosome 16 block 53815773–55761660, the FTO locus for BMI (ukb-b-19953, N≈462k). Strong association signal (min p ≈ 7×10⁻²⁸).

For each block we:

  1. Load the standardised GWAS summary statistics and the LD reference panel.
  2. Match GWAS SNPs to the LD panel by SNP ID.
  3. Randomly mask 30 % of matched variants (set z-scores to NaN).
  4. Impute the masked values using the PCA + elastic-net method implemented in pleiodb.impute.
  5. Compare imputed z-scores to the held-out truth.

def load_block(gwas_path: Path, block_str: str):
    """
    Load GWAS + LD panel for one block, inner-join on SNP ID, return a dict with:
      z_full, eaf, ld_sub, block_name, n_gwas, n_panel, n_matched
    """
    chrom, block_name = block_str.split("/", 1)
    tsv_path = LD_PANEL / chrom / f"{block_name}.tsv"
    ld_gz    = LD_PANEL / chrom / f"{block_name}.unphased.vcor1.gz"

    gwas  = pd.read_csv(gwas_path, sep="\t")
    panel = pd.read_csv(tsv_path, sep="\t")

    # Canonical SNP ID: strip "chr" prefix if present
    gwas["snp_bare"]  = gwas["SNP"].str.lstrip("chr")
    panel["snp_bare"] = panel["SNP"].str.lstrip("chr")

    # Inner join
    merged = gwas.merge(panel.reset_index().rename(columns={"index": "ld_row"}),
                        on="snp_bare", how="inner")

    ld_row_indices = merged["ld_row"].tolist()
    ld_sub = _load_ld_submatrix(ld_gz, ld_row_indices)

    return dict(
        z_full     = merged["Z"].values.astype(np.float64),
        eaf        = merged["EAF_x"].values.astype(np.float64),
        ld_sub     = ld_sub,
        block_name = block_name,
        n_gwas     = len(gwas),
        n_panel    = len(panel),
        n_matched  = len(merged),
    )


def run_benchmark(block_data: dict, mask_frac: float = MASK_FRAC, seed: int = SEED,
                  thresh: float = LD_THRESH):
    """
    Mask mask_frac of z-scores, impute, return results dict.
    """
    z_full = block_data["z_full"]
    ld_sub = block_data["ld_sub"]
    n = len(z_full)

    rng = np.random.default_rng(seed)
    mask_idx = rng.choice(n, size=int(n * mask_frac), replace=False)

    z_masked = z_full.copy()
    z_masked[mask_idx] = np.nan

    # PCA
    eigenvalues, eigenvectors = _ld_pca(ld_sub, thresh=thresh)
    n_comp = len(eigenvalues)

    # Elastic net imputation
    z_pred = _elastic_net_impute(z_masked, eigenvectors, n_comp)
    if z_pred is None:
        return None

    # Polynomial rescaling
    z_adj, corr = _poly_rescale(z_masked, z_pred, npoly=3)

    truth    = z_full[mask_idx]
    imputed  = z_adj[mask_idx]

    rmse = float(np.sqrt(np.mean((truth - imputed) ** 2)))
    slope, intercept = np.polyfit(truth, imputed, 1)
    r2 = corr ** 2 if np.isfinite(corr) else float("nan")

    return dict(
        truth     = truth,
        imputed   = imputed,
        z_masked  = z_masked,
        z_adj     = z_adj,
        corr      = corr,
        r2        = r2,
        rmse      = rmse,
        slope     = slope,
        intercept = intercept,
        n_comp    = n_comp,
        n_masked  = len(mask_idx),
        mask_idx  = mask_idx,
    )

Scatter plots: imputed vs truth

fig, axes = plt.subplots(1, 2, figsize=(11, 5))

for ax, res, title in zip(
    axes,
    [null_res, signal_res],
    ["Null region (chr1 168–170 Mb)", "Signal region (FTO, chr16 53.8–55.8 Mb)"]
):
    truth   = res["truth"]
    imputed = res["imputed"]

    # Density scatter
    xy = np.vstack([truth, imputed])
    try:
        from scipy.stats import gaussian_kde
        z = gaussian_kde(xy)(xy)
        idx = z.argsort()
        sc = ax.scatter(truth[idx], imputed[idx], c=z[idx], s=6, cmap="viridis",
                        rasterized=True)
        plt.colorbar(sc, ax=ax, label="Density")
    except Exception:
        ax.scatter(truth, imputed, s=4, alpha=0.3, color="steelblue")

    # Identity line
    lim = max(abs(truth).max(), abs(imputed).max()) * 1.05
    ax.axline((0, 0), slope=1, color="tomato", linewidth=1.2, linestyle="--", label="y=x")

    # OLS fit line
    x_line = np.linspace(-lim, lim, 200)
    ax.plot(x_line, res["slope"] * x_line + res["intercept"],
            color="orange", linewidth=1.2, label=f"OLS (slope={res['slope']:.2f})")

    ax.set_xlim(-lim, lim)
    ax.set_ylim(-lim, lim)
    ax.set_xlabel("True z-score")
    ax.set_ylabel("Imputed z-score")
    ax.set_title(title, fontsize=10)
    ax.legend(fontsize=8)
    ax.text(0.05, 0.93, f"r = {res['corr']:.3f}\nRMSE = {res['rmse']:.3f}",
            transform=ax.transAxes, fontsize=9, verticalalignment="top",
            bbox=dict(boxstyle="round", facecolor="wheat", alpha=0.5))

Interpretation:

  • In the null region (no true signal), imputation recovers z-scores with modest correlation. The elastic-net model captures the LD structure well enough to reproduce the background noise pattern, though RMSE is non-trivial because z-scores are small.

  • In the signal region (FTO locus), imputation is substantially better: the large z-scores at the association peak are reproduced faithfully because nearby LD proxies carry almost the same information. This is exactly the setting where imputation is most valuable for pleiodb — a dense association signal that was not tested in a sparse array-based study.

  • Correlation degrades gracefully as the mask fraction increases (see Section 7), but remains useful even at 60–70% masking in the signal region.
